28           environments：配置多个连接数据库的环境

30             default：设置默认使用的环境的id 
31        -->
32       <environments default="development">
33       <!--
34           environment：配置某个具体的环境 
35           属性： 
36               id:表示连接数据库的环境的唯一标识，不能重复 
37    
38       -->
39           <environment id="development">
40               <!--  transactionManager：设置事务管理方式  -->
41               <!--属性： 
42                       type="JDBC|MANAGED" 
43                       JDBC：表示当前环境中，执行SQL时，使用的是JDBC中原生的事物管理方式，事物的提交或回滚需要手动处理 
44                       MANAGED：被管理，例如Spring 
45               -->
46               <transactionManager type="JDBC"/>
47               <!--
48                   dataSource：配置数据源 
49                   属性： 
50                       type：设置数据源的类型 
51                       type：“POOLED|UNPOOLED|JNDI“ 
52                       POOLED：表示使用数据库池缓存数据库连接 
53                       UNPOOLED：表示不使用数据库连接池 
54                       JNDI：表示使用上下文的数据源 
55               -->
56               <dataSource type="POOLED">
57                   <!--设置连接数据库的驱动-->
58                   <property name="driver" value="${jdbc.driver}"/>
59                   <!--设置连接数据库的连接地址-->
60                   <property name="url" value="${jdbc.url}"/>
61                   <!--设置连接数据库的用户名-->
62                   <property name="username" value="${jdbc.username}"/>
63                   <!--设置连接数据库的密码-->
64                   <property name="password" value="${jdbc.password}"/>
65               </dataSource>
66           </environment>
67       </environments>
